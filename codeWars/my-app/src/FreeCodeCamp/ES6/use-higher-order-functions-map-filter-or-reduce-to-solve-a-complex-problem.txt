const squareList = arr => {
  // Only change code below this line
  let signFloatAArr= arr.filter(item=>Number.isInteger(item)&&Math.sign(item)==1);
  let squareSignFloatAArr=signFloatAArr.map(num=>num*num)
  return squareSignFloatAArr;
  // Only change code above this line
};

const squaredIntegers = squareList([-3, 4.8, 5, 3, -3.2]);
