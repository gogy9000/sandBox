
function sumAll(arr) {


let max= Math.max.apply(0,arr)
let min= Math.min.apply(0,arr)
let count =Math.min.apply(0,arr)
 const recursia=(min,max)=>{
   if(min>=max){
     return
   }else{
     count+=min+1;
      recursia(min+1,max)
 }
 }
recursia(min,max)
return count
}

console.log(sumAll([1,4]));
