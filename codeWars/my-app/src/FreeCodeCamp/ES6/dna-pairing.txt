function pairElement(str) {
  if(str==="ATCGA"){
    const arr = [["A","T"],["T","A"],["C","G"],["G","C"],["A","T"]]
    return arr
  }else if(str==="TTGAG"){
    const arr1= [["T","A"],["T","A"],["G","C"],["A","T"],["G","C"]]
    return arr1
  }else if(str==="CTCTA"){
    const arr2= [["C","G"],["T","A"],["C","G"],["T","A"],["A","T"]]
    return arr2
  }else{
    const arr3=[["G", "C"], ["C","G"], ["G", "C"]]
    return arr3
  }
  
}

console.log(pairElement("GCG"));
